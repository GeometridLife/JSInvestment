```python
        peak = window.cummax()
        dd = (peak - window) / peak
        forward_mdd.append(dd.max())

    # 2. Forward MDD (조건부)

    current_dd = (running_peak - prices) / running_peak
    bins = [(0, 0.05), (0.05, 0.10), (0.10, 0.20), (0.20, 1.0)]

    # DD 이벤트 시작(전고점 이탈) → 전고점 복귀까지 영업일
    # → 중앙값

    n_years = len(prices) / 252
```

```python
def calc_macd_strategy(prices, c=0.0002):
    ema12 = prices.ewm(span=12, adjust=False).mean()
    ema26 = prices.ewm(span=26, adjust=False).mean()
    macd_line = ema12 - ema26
    signal = macd_line.ewm(span=9, adjust=False).mean()
    hist = macd_line - signal
    
    # 골든크로스: hist[t-1] <= 0 and hist[t] > 0 → 진입
    # 데드크로스: hist[t-1] >= 0 and hist[t] < 0 → 청산
    
    trades = []
    in_position = False
    for t in range(1, len(hist)):
        if not in_position and hist.iloc[t-1] <= 0 and hist.iloc[t] > 0:
            entry_price = prices.iloc[t]
            in_position = True
        elif in_position and hist.iloc[t-1] >= 0 and hist.iloc[t] < 0:
            exit_price = prices.iloc[t]
            log_return = np.log(exit_price / entry_price) - 2*c
            trades.append(log_return)
            in_position = False
    
    return {
        'cum_return': sum(trades),
        'win_rate': sum(1 for t in trades if t > 0) / len(trades) if trades else 0,
        'avg_return': np.mean(trades) if trades else 0,
        'n_trades': len(trades),
    }
```

```python
def calc_high52_strategy(prices, c=0.0002, trailing_stop=0.05, min_gap=5):
    high_250 = prices.rolling(250, min_periods=250).max()
    
    trades = []
    in_position = False
    last_entry_idx = -min_gap
    peak_since_entry = 0
    
    for t in range(250, len(prices)):
        if not in_position:
            # 진입: 신고가 갱신 + 최소 간격
            if prices.iloc[t] >= high_250.iloc[t] and (t - last_entry_idx) >= min_gap:
                entry_price = prices.iloc[t]
                peak_since_entry = entry_price
                in_position = True
                last_entry_idx = t
        else:
            # 트레일링 스톱 업데이트
            peak_since_entry = max(peak_since_entry, prices.iloc[t])
            # 청산: 고점 대비 -5%
            if prices.iloc[t] < peak_since_entry * (1 - trailing_stop):
                exit_price = prices.iloc[t]
                log_return = np.log(exit_price / entry_price) - 2*c
                trades.append(log_return)
                in_position = False
    
    # 미청산 포지션 처리: 마지막 가격으로 청산
    if in_position:
        log_return = np.log(prices.iloc[-1] / entry_price) - 2*c
        trades.append(log_return)
    
    return {
        'cum_return': ...,
        'win_rate': ...,
        'avg_return': ...,
        'n_trades': ...,
    }
```
